208 public virtual ITween Advance(float delta)
209 {
210 if (m_previous != null && !m_previous.done)
211 {
212 m_time = 0f;
213 }
214 else
215 {
216 m_time += (delta * m_timescale);
217 m_time = clamp(m_time, 0f, m_duration);
218 }
219
220 m_timeClamped = clamp(m_time - m_delay, 0f, m_duration);
221 m_progress = m_timeClamped / m_duration;
222 m_val = InternalComputeValue(m_timeClamped);
223 return this;
