74 #region Expo

85 [MethodImpl(MethodImplOptions.AggressiveInlining)]
86 public static float ExpoEaseOut(float t, float d, float b = 0f, float c = 1f)
87 {
88 return (t == d) ? b + c : c * (-pow(2.0f, -10.0f * t / d) + 1.0f) + b;
89 }
90
100 [MethodImpl(MethodImplOptions.AggressiveInlining)]
101 public static float ExpoEaseIn(float t, float d, float b = 0f, float c = 1f)
102 {
103 return (t == 0) ? b : c * pow(2.0f, 10.0f * (t / d - 1.0f)) + b;
104 }
105
115 [MethodImpl(MethodImplOptions.AggressiveInlining)]
116 public static float ExpoEaseInOut(float t, float d, float b = 0f, float c = 1f)
117 {
118 if (t == 0)
119 return b;
120
121 if (t == d)
122 return b + c;
123
124 if ((t /= d * 0.5f) < 1.0f)
125 return c * 0.5f * pow(2.0f, 10.0f * (t - 1.0f)) + b;
126
127 return c * 0.5f * (-pow(2.0f, -10.0f * --t) + 2.0f) + b;
128 }
129
130
140 [MethodImpl(MethodImplOptions.AggressiveInlining)]
141 public static float ExpoEaseOutIn(float t, float d, float b = 0f, float c = 1f)
142 {
143 if (t < d * 0.5f)
144 return ExpoEaseOut(t * 2.0f, b, c * 0.5f, d);
145
146 return ExpoEaseIn((t * 2.0f) - d, b + c * 0.5f, c * 0.5f, d);
147 }

162 [MethodImpl(MethodImplOptions.AggressiveInlining)]
163 public static float CircEaseOut(float t, float d, float b = 0f, float c = 1f)
164 {
165 return c * sqrt(1.0f - (t = t / d - 1.0f) * t) + b;

177 [MethodImpl(MethodImplOptions.AggressiveInlining)]
178 public static float CircEaseIn(float t, float d, float b = 0f, float c = 1f)
179 {
180 return -c * (sqrt(1.0f - (t /= d) * t) - 1.0f) + b;

193 public static float CircEaseInOut(float t, float d, float b = 0f, float c = 1f)

582 [MethodImpl(MethodImplOptions.AggressiveInlining)]
583 public static float ElasticEaseOut(float t, float d, float b = 0f, float c = 1f)
584 {
585 if ((t /= d) == 1)
586 return b + c;
587
588 float p = d * .3f;
589 float s = p / 4.0f;
590
591 return (c * pow(2.0f, -10.0f * t) * sin((t * d - s) * TAU / p) + c + b);
592 }
593
603 [MethodImpl(MethodImplOptions.AggressiveInlining)]
604 public static float ElasticEaseIn(float t, float d, float b = 0f, float c = 1f)
605 {
606 if ((t /= d) == 1.0f)
607 return b + c;
608
609 float p = d * .3f;
610 float s = p / 4.0f;
611
612 return -(c * pow(2.0f, 10.0f * (t -= 1.0f)) * sin((t * d - s) * TAU / p)) + b;
613 }
614
624 [MethodImpl(MethodImplOptions.AggressiveInlining)]
625 public static float ElasticEaseInOut(float t, float d, float b = 0f, float c = 1f)
626 {
627 if ((t /= d * 0.5f) == 2.0f)
628 return b + c;
629
630 float p = d * (.3f * 1.5f);
631 float s = p / 4.0f;
632
633 if (t < 1)
634 return -.5f * (c * pow(2.0f, 10.0f * (t -= 1.0f)) * sin((t * d - s) * TAU / p)) + b;
635 return c * pow(2.0f, -10.0f * (t -= 1)) * sin((t * d - s) * TAU / p) * .5f + c + b;
636 }
637
638
648 [MethodImpl(MethodImplOptions.AggressiveInlining)]
649 public static float ElasticEaseOutIn(float t, float d, float b = 0f, float c = 1f)
650 {
651 if (t < d / 2)
652 return ElasticEaseOut(t * 2, b, c / 2, d);
653 return ElasticEaseIn((t * 2) - d, b + c / 2, c / 2, d);

669 [MethodImpl(MethodImplOptions.AggressiveInlining)]
670 public static float BounceEaseOut(float t, float d, float b = 0f, float c = 1f)
671 {
672 if ((t /= d) < (1 / 2.75))
673 return c * (7.5625f * t * t) + b;
674 else if (t < (2.0f / 2.75f))
675 return c * (7.5625f * (t -= (1.5f / 2.75f)) * t + .75f) + b;
676 else if (t < (2.5f / 2.75f))
677 return c * (7.5625f * (t -= (2.25f / 2.75f)) * t + .9375f) + b;
678 else
679 return c * (7.5625f * (t -= (2.625f / 2.75f)) * t + .984375f) + b;
680 }
681
691 [MethodImpl(MethodImplOptions.AggressiveInlining)]
692 public static float BounceEaseIn(float t, float d, float b = 0f, float c = 1f)
693 {
694 return c - BounceEaseOut(d - t, 0, c, d) + b;
695 }

706 [MethodImpl(MethodImplOptions.AggressiveInlining)]
707 public static float BounceEaseInOut(float t, float d, float b = 0f, float c = 1f)
